rows = []
task_columns = [f"Round {i+1}" for i in range(6)]

for task in TASKS:
    for condition in CONDITIONS:
        PIDS = ALL_PIDS[condition]
        for pid in PIDS:
            basepath = os.path.join(condition, pid)  # basepath for PID
            df = pd.read_csv(os.path.join(basepath, 'selections.csv'))

            ### NUMBER OF TIMES RECOMMENDATION WAS USED BEFORE SUBMISSION ###
            mask = (df["appMode"] == task)
            selections = df[mask]['recommendationShown'].replace({True: "# Yes", False: "# No"}).tolist()

            # Add to list of rows to create DataFrame from
            rows.append([pid, task, condition, *selections])

# Combine rows
df_get_rec = pd.DataFrame(
    rows,

rows = []
task_columns = [f"Round {i+1}" for i in range(6)]

dfs_all = []


### BY CONDITION ###

for condition in CONDITIONS:
    dfs_condition = []
    PIDS = ALL_PIDS[condition]
    
    for task in TASKS:
        for pid in PIDS:
            basepath = os.path.join(condition, pid)  # basepath for PID
            df = pd.read_csv(os.path.join(basepath, 'selections.csv'))

            ### Count of matching answers grouped by recommendation shown ###
            mask = (df["appMode"] == task)
            matching = df[mask].assign(matching=lambda x: x["selectedId"] == x["botChoice"])
            matching_count = matching.groupby(["recommendationShown", "matching"]).size()
            dfs_condition.append(matching_count)
            dfs_all.append(matching_count)
            
    # Combine rows
    print(f"========== {condition} ==========")
    df_matching = pd.concat(dfs_condition, axis=1).fillna(0.0)

    ### GET SUM/AVG OF MATCHING ###
    sum_matching = df_matching.sum(axis=1)
    print(sum_matching)
    avg_matching = df_matching.mean(axis=1)
    print(avg_matching)
    print()

    ### PLOT ###
    labels = [
        "Recommendation Not Shown / Not Matching",
        "Recommendation Not Shown / Matching",
        "Recommendation Shown / Not Matching",
        "Recommendation Shown / Matching",
    ]
    data = [
        df_matching.iloc[0].tolist(),
        df_matching.iloc[1].tolist(),
        df_matching.iloc[2].tolist(),
        df_matching.iloc[3].tolist(),
    ]
    fig, ax = plt.subplots(figsize=(4,3))
    bp = ax.boxplot(data, vert=False, labels=labels)
    ax.set_xlim(-0.1, 6.1)
    ax.set_xlabel("Number of rounds")
    ax.xaxis.grid(True, linestyle='-', which='major', color='lightgrey', alpha=0.5)
    ax.set_title(f"{condition} CONDITION")
